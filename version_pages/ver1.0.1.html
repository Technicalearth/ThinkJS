Samples of Think JS Version 1.0.1, Try it in console.


// To add html in think js

think("button").html("<b>Hello, World</b>");

// To add text in a element

think("button").text("Hello World");

// To see the html in console

console.log(think("button").html());

// To see the text in console

console.log(think("button").text());

// To add events on an element

think("button").on("click", function(){
    alert("Hello World");
});

// To add css in an element

think("button").css({
    color: "red",
    "fontSize": "33px"
});

// To add a click event

think("button").click(function(){
    alert("Hello, World!");
});

// To add a mouseover event

think("button").mouseover(function(){
    alert("Hello, World!");
});

// To add a class name

think("button").addclass("btn");

// To remove a class name

think("button").removeclass("btn");
